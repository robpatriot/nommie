# Database Configuration
# Copy this file to .env and set secure passwords

# PostgreSQL container settings
POSTGRES_PORT=5432
POSTGRES_DB=nommie
POSTGRES_USER=postgres
POSTGRES_PASSWORD=your_postgres_password_here

# Application database passwords
APP_DB_PASSWORD=your_app_password_here
NOMMIE_OWNER_PASSWORD=your_owner_password_here

# Application database URL (app role - used by main.rs)
# This should point to the app role, not the owner role
DATABASE_URL=postgres://nommie_app:your_app_password_here@localhost:5432/nommie

# Frontend environment variables
NEXT_PUBLIC_APP_NAME=Nommie
